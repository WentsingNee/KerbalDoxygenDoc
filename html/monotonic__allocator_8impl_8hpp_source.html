136 void monotonic_allocator<T, UpstreamAllocator>::fill_buffer(size_type next_pure_buffer_size)
137 {
138 typedef kerbal::memory::allocator_traits<upstream_allocator> allocator_traits;
139 typedef auto_node U;
140
141 std::size_t const basic_align = allocator_traits::minimum_alignment(upstream_alloc());
142 std::size_t gross_buffer_size =
143 kerbal::memory::alignment_maximum_offset(KERBAL_ALIGNOF(U), basic_align) +
144 sizeof(U) +
145 kerbal::memory::alignment_maximum_offset(ALIGNOF_T::value, KERBAL_ALIGNOF(U)) +
146 next_pure_buffer_size
147 ;
148
149 void * const p_gross = allocator_traits::allocate(upstream_alloc(), gross_buffer_size);
150 std::size_t p_gross_sz = reinterpret_cast<std::size_t>(p_gross);
151
152 std::size_t pu_sz = kerbal::memory::align_ceil(p_gross_sz, KERBAL_ALIGNOF(U));
153 std::size_t pt_sz = kerbal::memory::align_ceil(pu_sz + sizeof(U), ALIGNOF_T::value);
154
155 U * pu = reinterpret_cast<U *>(pu_sz);
156 kerbal::memory::construct_at(pu, kerbal::utility::in_place_t(), p_gross, gross_buffer_size);
157 this->k_list.push_front(*pu);
158
159 this->k_current_pure_buffer_head = reinterpret_cast<pointer>(pt_sz);
160 this->k_current_used_in_bytes = 0;
161 this->k_current_buffer_end = reinterpret_cast<pointer>(pt_sz + next_pure_buffer_size);
162 }

172 template <typename T, typename UpstreamAllocator>
173 typename
174 monotonic_allocator<T, UpstreamAllocator>::pointer
175 monotonic_allocator<T, UpstreamAllocator>::allocate(size_type n, kerbal::memory::align_val_t align)
176 {
177 if (n > MAX_SIZE::value) {
178 kerbal::utility::throw_this_exception_helper<kerbal::memory::bad_array_new_length>::throw_this_exception();
179 }
180 std::size_t pure_size = n * sizeof(value_type);
181 char * pstart = reinterpret_cast<char *>(this->k_current_pure_buffer_head) + this->k_current_used_in_bytes;
182 char * phead = align_ceil(pstart, align.val);
183 char * pend = phead + pure_size;
184 if (pend <= reinterpret_cast<char *>(this->k_current_buffer_end)) {
185 this->k_current_used_in_bytes = pend - reinterpret_cast<char *>(this->k_current_pure_buffer_head);
186 return reinterpret_cast<pointer>(phead);
187 }
188 size_type next_pure_buffer_size = this->get_next_pure_buffer_size();
189 if (static_cast<size_type>(pend - pstart) > next_pure_buffer_size) {
190 next_pure_buffer_size = static_cast<size_type>(pend - pstart);
191 }
192 fill_buffer(next_pure_buffer_size);
193 pstart = reinterpret_cast<char *>(this->k_current_pure_buffer_head) + this->k_current_used_in_bytes;
194 phead = align_ceil(pstart, align.val);
195 pend = phead + pure_size;
196 this->k_current_used_in_bytes = pend - reinterpret_cast<char *>(this->k_current_pure_buffer_head);
197 return reinterpret_cast<pointer>(phead);
198 }
