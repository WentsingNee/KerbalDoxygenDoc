96 template <typename ResultType, ResultType a, ResultType m>
97 struct static_mul_mod_impl_dispatch<ResultType, a, m, 0> // m != 0 && a % m != 0
98 {
99 private:
100 KERBAL_STATIC_ASSERT(m != 0 && a % m != 0 && a % m == a, "static check failed");
101
102 KERBAL_CONSTEXPR14
103 static ResultType cacl_impl(ResultType s, kerbal::type_traits::type_identity<void>) KERBAL_NOEXCEPT
104 {
105 if (s <= kerbal::numeric::numeric_limits<ResultType>::MAX::value / a) {
106 return a * s % m;
107 }
108
109 /* Schrage's algorithm
110 *
111 * let m = a * q + r (q and r are all integers, obviously q = m / a, r = m % a)
112 * a * s = a * (q * (s / q) + s % q)
113 * = a * q * (s / q) + a * (s % q)
114 * = (m - r) * (s / q) + a * (s % q)
115 * = a * (s % q) - r * (s / q) (mod m)
116 * when r <= q, both terms are less than m so it does not overflow.
117 */
118 typedef kerbal::type_traits::integral_constant<ResultType, m / a> Q;
119 typedef kerbal::type_traits::integral_constant<ResultType, m % a> R;
120
121 ResultType t0 = a * (s % Q::value);
122 ResultType t1 =
123 R::value <= Q::value ?
124 R::value * (s / Q::value) :
125 static_mul_mod_impl<ResultType, R::value, m>::cacl(s / Q::value);
126 return t0 - t1 + (t0 < t1) * m;
127 }
128
129 template <typename T>
130 KERBAL_CONSTEXPR14
131 static ResultType cacl_impl(ResultType s, kerbal::type_traits::type_identity<T>) KERBAL_NOEXCEPT
132 {
133 // a * (m - 1) will not overflow under T
134 return T(a) * T(s) % m;
135 }
136
137 public:
138 // a %= m, s %= m
139 KERBAL_CONSTEXPR14
140 static ResultType cacl(ResultType s) KERBAL_NOEXCEPT
141 {
142 typedef typename find_never_overflow_integer_type_for_mul_mod<ResultType, a, m>::type T;
143 return cacl_impl(s, kerbal::type_traits::type_identity<T>());

335 // a ** (n - 1) + ... + a ** 3 + a ** 2 + a ** 1 + 1
336 template <typename ResultType, ResultType a, ResultType m>
337 struct sigma_pow_mod
338 {
339 KERBAL_CONSTEXPR14
340 static ResultType cacl(unsigned long long n) KERBAL_NOEXCEPT
341 {
342 /*
343 * e.g. for n = 21
344 *
345 * 16 8 4 2 1
346 * 1 0 1 0 1 r = r * a + 1
347 * 1 0 1 0 r = r * (a ** 10 + 1)
348 * 1 0 1 r = r * (a ** 5 + 1)
349 * 1 0 0 r = r * a + 1
350 * 1 0 r = r * (a ** 2 + 1)
351 * 1 r = r * a + 1
352 */
353
354 // pre: n != 0
355 int lead = CHAR_BIT * sizeof(n) - 1;
356 {
357 // to opt: lead = CHAR_BIT * sizeof(n) - 1 - __builtin_clzll(n);
358 while (((n >> lead) & 1) == 0) {
359 lead--;
360 }
361 }
362 ResultType r = 1;
363 bool b = true;
364 ResultType poa = 1;
365 while (lead > 0) {
366 --lead;
367 poa = partial_static_mul_mod<ResultType, m>::cacl(poa, poa);
368 if (b) {
369 poa = partial_static_mul_mod<ResultType, m>::cacl(a, poa);
370 }
371 r = partial_static_mul_mod<ResultType, m>::cacl(
372 r,
373 add_mod<ResultType, m>::cacl(poa, 1)
374 );
375 b = (n >> lead) & 1;
376 if (b) {
377 // r = r * a + 1
378 r = static_mul_mod<ResultType, a, m>::cacl(r);
379 r = add_mod<ResultType, m>::cacl(r, 1);
380 }
381 }
382 return r;
383 }

389 template <typename ResultType, ResultType a, ResultType c, ResultType m>
390 class linear_congruential_engine
391 {
392 private:
393 KERBAL_STATIC_ASSERT(
394 m == 0 || (a < m && c < m),
395 "If the template parameter m is not 0, the following relations shall hold: a < m and c < m"
396 );
397
398 public:
399 typedef ResultType result_type;
400
401 typedef kerbal::type_traits::integral_constant<result_type, a> MULTIPLIER;
402 typedef kerbal::type_traits::integral_constant<result_type, c> INCREMENT;
403 typedef kerbal::type_traits::integral_constant<result_type, m> MODULUS;
404
405 typedef kerbal::type_traits::integral_constant<result_type, 1u> DEFAULT_SEED;
406
407 private:
408 result_type state_value;
409
410 public:
411
412 KERBAL_CONSTEXPR
413 explicit
414 linear_congruential_engine(const result_type & seed = DEFAULT_SEED::value) KERBAL_NOEXCEPT :
415 state_value(
416 m == 0u ?
417 (seed == 0u ? 1u : seed) :
418 (seed % m == 0u ? 1u : seed)
419 )
420 {
421 }
422
423 KERBAL_CONSTEXPR14
424 result_type operator()() KERBAL_NOEXCEPT
425 {
426 if (m == 0) {
427 state_value = a * state_value + c;
428 } else {
429 state_value = detail::static_mul_mod<result_type, a, m>::cacl(state_value);
430 state_value = detail::add_mod<result_type, m>::cacl(state_value, c);
431 }
432 return state_value;
433 }
434
435 KERBAL_CONSTEXPR14
436 void discard(unsigned long long times) KERBAL_NOEXCEPT
437 {
438 if (times == 0) {
439 return ;
440 }
441 /*
442 * s_{i+n} = a^{n} \times s_i + c \times \sum_{i=0}^{n-1} a^i
443 * = a ** n * s_i + c * (a ** (n - 1) + a ** (n - 2) + ... + a + 1)
444 */
445 result_type a_pow_times_mod_m = detail::pow_mod<result_type, a, m>::cacl(times);
446 state_value = detail::partial_static_mul_mod<ResultType, m>::cacl(a_pow_times_mod_m, state_value);
447 if (c != 0) {
448 if (a == 1) {
449 result_type c_mul_times_mod_m = detail::static_mul_mod<result_type, c, m>::cacl(times);
450 state_value = detail::add_mod<result_type, m>::cacl(state_value, c_mul_times_mod_m);
451 } else {
452 result_type sigma = detail::sigma_pow_mod<result_type, a, m>::cacl(times);
453 sigma = detail::static_mul_mod<result_type, c, m>::cacl(sigma);
454 state_value = detail::add_mod<result_type, m>::cacl(state_value, sigma);
455 }
456 }
457 }

463 m == 0u ?
464 (seed == 0u ? 1u : seed) :
465 (seed % m == 0u ? 1u : seed)
