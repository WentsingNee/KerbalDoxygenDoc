79 /* Schrage's algorithm
80 *
81 * let m = a * q + r (q and r are all integers, obviously q = m / a, r = m % a)
82 * a * b = a * (q * (b / q) + b % q)
83 * = a * q * (b / q) + a * (b % q)
84 * = (m - r) * (b / q) + a * (b % q)
85 * = a * (b % q) - r * (b / q) (mod m)
86 * when r <= q, both terms are less than m so it does not overflow.
87 */

90 ResultType t1 = schrage_t1_impl(b, kerbal::type_traits::bool_constant<R::value <= Q::value>());

92 }
