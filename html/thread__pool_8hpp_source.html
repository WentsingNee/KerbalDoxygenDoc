172 void operator()() const
173 {
174 while (!self->k_stopped.load()) {
175 std::unique_lock<std::mutex> lock(self->k_mutex);
176 self->k_cv.wait(lock, [self = this->self] {
177 return self->k_stopped.load() || !self->k_jobs_queue.empty();
178 });
179 if (self->k_stopped.load() && self->k_jobs_queue.empty()) {
180 return;

182 job_type job(kerbal::compatibility::move(self->k_jobs_queue.front()));
183 self->k_jobs_queue.pop();
184 lock.unlock();
185 --self->k_idle_threads_num;

188 }
189 }
