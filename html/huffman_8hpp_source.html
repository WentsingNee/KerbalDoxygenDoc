156 {
157 char code_buff[256 + 1] = {}; // code_buff start from index 1, doesn't occur '\0'
158 std::size_t code_length = 0;
159
160 kerbal::container::static_vector<kerbal::utility::compressed_pair<short, std::size_t>, 255> st;
161 st.emplace_back_unsafe(tree_root, code_length);
162
163 do {
164 tree_root = st.back().first();
165 code_length = st.back().second();
166 st.pop_back_unsafe();
167 code_buff[code_length++] = '1';
168
169 while (!base[tree_root].is_leaf()) {
170 // in huffman coding-tree, if node is non-leaf, base[node].right is always not NULL
171 // <=> traverse(base[node].right, code_buff, code_length + 1);
172 st.emplace_back_unsafe(base[tree_root].right, code_length);
173
174 // <=> traverse(base[node].left, code_buff, code_length + 1);
175 tree_root = base[tree_root].left;
176 code_buff[code_length++] = '0';
177 }
178 unsigned char value = base[tree_root].value;
179 kerbal::algorithm::copy(code_buff + 1, code_buff + code_length, this->codes_table[value]);
180 this->codes_table[value][code_length] = '\0';
181 } while (!st.empty());

184 KERBAL_CONSTEXPR14
185 void build_huffman_table(const huffman_alphabet & alphabet) KERBAL_NOEXCEPT
186 {
187 using detail::huffman_tree_node;
188
189 kerbal::container::static_vector<huffman_tree_node, 256 * 2 - 1> huffman_tree_nodes;
190 for (std::size_t i = 0; i < 256; ++i) {
191 std::size_t cnt = alphabet.cnt[i];
192 if (cnt == 0) {
193 continue;
194 }
195 huffman_tree_nodes.emplace_back_unsafe(static_cast<unsigned char>(i), cnt);
196 }
197
198 const huffman_tree_node * base = &huffman_tree_nodes[0];
199 short tree_root = -1;
200 {
201 short ptr2nodes[256 * 2 - 1] = {};
202 short ptr2nodes_cnt = static_cast<short>(huffman_tree_nodes.size());
203 kerbal::algorithm::iota(ptr2nodes + 0, ptr2nodes + ptr2nodes_cnt, 0);
204
205 using detail::huffman_heap_cmp;
206 huffman_heap_cmp heap_cmp(base);
207 kerbal::algorithm::make_heap(ptr2nodes + 0, ptr2nodes + ptr2nodes_cnt, heap_cmp);
208 while (ptr2nodes_cnt != 1) {
209 kerbal::algorithm::pop_heap(ptr2nodes + 0, ptr2nodes + ptr2nodes_cnt, heap_cmp);
210 --ptr2nodes_cnt;
211 short a = ptr2nodes[ptr2nodes_cnt];
212
213 kerbal::algorithm::pop_heap(ptr2nodes + 0, ptr2nodes + ptr2nodes_cnt, heap_cmp);
214 --ptr2nodes_cnt;
215 short b = ptr2nodes[ptr2nodes_cnt];
216
217 huffman_tree_nodes.emplace_back_unsafe(a, b, base[a].weight + base[b].weight);
218 ptr2nodes[ptr2nodes_cnt++] = static_cast<short>(huffman_tree_nodes.size() - 1);
219 kerbal::algorithm::push_heap(ptr2nodes + 0, ptr2nodes + ptr2nodes_cnt, heap_cmp);
220 }
221 tree_root = ptr2nodes[0];
222 }
223
224 traverse_tree_gen_table(base, tree_root);
225 }
