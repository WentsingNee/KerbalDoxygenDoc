289 KERBAL_CONSTEXPR20
290 typename kerbal::type_traits::enable_if<
291 kerbal::iterator::is_forward_compatible_iterator<ForwardIterator>::value,
292 iterator
293 >::type
294 insert(const_iterator pos, ForwardIterator first, ForwardIterator last)
295 {
296 if (first == last) {
297 return pos.cast_to_mutable();
298 }
299
300 ForwardIterator it(first);
301 ForwardIterator next(first); ++next;
302
303 while (next != last) {
304 (*it).next = &*next;
305 (*next).prev = &*it;
306 it = next;
307 ++next;
308 }
309
310 list_type_unrelated::k_hook_node(pos, &*first, &*it);
311 return iterator(&*first);
