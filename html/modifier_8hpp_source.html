302 template <typename InputIterator1, typename InputIterator2, typename OutputIterator, typename CompareFuntion>
303 KERBAL_CONSTEXPR14
304 OutputIterator
305 merge(InputIterator1 a_first, InputIterator1 a_last,
306 InputIterator2 b_first, InputIterator2 b_last,
307 OutputIterator to, CompareFuntion cmp)
308 {
309 while (a_first != a_last) {
310 if (b_first != b_last) {
311 if (cmp(*b_first, *a_first)) { // b < a
312 kerbal::operators::generic_assign(*to, *b_first); // *to = *b_first;
313 ++to; // advance `to` before `b_first` and `a_first`
314 // in case of the advance of `b_first` or `a_first` throws exception
315 ++b_first;
316 } else { // b >= a
317 kerbal::operators::generic_assign(*to, *a_first); // *to = *a_first;
318 ++to;
319 ++a_first;
320 }
321 } else {
322 return kerbal::algorithm::copy(a_first, a_last, to);
323 }
324 }
325 return kerbal::algorithm::copy(b_first, b_last, to);
326 }

469 rotate_swap(ForwardIterator & first, ForwardIterator n_first, ForwardIterator last)

487 template <typename ForwardIterator>
488 KERBAL_CONSTEXPR14
489 ForwardIterator
490 rotate(ForwardIterator first, ForwardIterator n_first, ForwardIterator last, std::forward_iterator_tag)
491 {
492 typedef ForwardIterator iterator;
493
494 if (n_first == first) {
495 return last;
496 }
497
498 if (n_first == last) {
499 return first;
500 }
501
502 iterator ret(rotate_swap(first, n_first, last));
503
504 n_first = first;
505 first = ret;
506
507 while (n_first != first && n_first != last) {
508 iterator left(rotate_swap(first, n_first, last));
509
510 n_first = first;
511 first = left;
512 // next loop: rotate(left, first, last)
513 }
514 return ret;
515 }

517 template <typename RandomAccessIterator>
518 KERBAL_CONSTEXPR14
519 RandomAccessIterator
520 rotate(RandomAccessIterator first, RandomAccessIterator n_first, RandomAccessIterator last,
521 std::random_access_iterator_tag)
522 {
523 typedef RandomAccessIterator iterator;
524 typedef typename kerbal::iterator::iterator_traits<iterator>::difference_type difference_type;
525 typedef typename kerbal::iterator::iterator_traits<iterator>::value_type value_type;
526
527 difference_type left_len(n_first - first);
528 difference_type right_len(last - n_first);
529
530 iterator ret(first + right_len);
531
532 while (left_len != 0 && right_len != 0) {
533
534// cout << left_len << " " << right_len << endl;
535
536 if (left_len <= right_len) {
537 if (left_len == 1) {
538 value_type t(kerbal::compatibility::to_xvalue(*first));
539 *kerbal::algorithm::move(n_first, last, first) = kerbal::compatibility::to_xvalue(t);
540 break;
541 }
542 while (n_first != last) {
543 kerbal::algorithm::iter_swap(first, n_first);
544 ++first;
545 ++n_first;
546 }
547 right_len %= left_len;
548 left_len -= right_len;
549 n_first = last - right_len;
550 // next loop: rotate(left, n_first', last)
551 } else {
552 if (right_len == 1) {
553 value_type t(kerbal::compatibility::to_xvalue(*n_first));
554 kerbal::algorithm::move_backward(first, n_first, last);
555 *first = kerbal::compatibility::to_xvalue(t);
556 break;
557 }
558 while (n_first != first) {
559 --n_first;
560 --last;
561 kerbal::algorithm::iter_swap(n_first, last);
562 }
563 left_len %= right_len;
564 right_len -= left_len;
565 n_first = first + left_len;
566 // next loop: rotate(first, n_first', right)
567 }
568 }
569
570 return ret;
571 }

597 template <typename ForwardIterator, typename BinaryPredict>
598 KERBAL_CONSTEXPR14
599 void _K_inplace_merge_fwd_iter_loop(ForwardIterator first, ForwardIterator mid, ForwardIterator last, BinaryPredict cmp)
600 {
601 typedef ForwardIterator iterator;
602
603 while (static_cast<bool>(first != mid) && static_cast<bool>(mid != last)) {
604
605 if (kerbal::iterator::next(first) == mid) {
606 while (mid != last) {
607 if (cmp(*mid, *first)) { // *first > *mid

610 ++mid;
611 } else {
612 break;
613 }
614 }
615 return;

621 _K_inplace_merge_fwd_iter_loop(first, lmid, mid, cmp);
622
623 first = mid;
624 mid = rmid; // kerbal::algorithm::inplace_merge(mid, rmid, last, cmp);
625 }

635 template <typename RandomAccessIterator, typename BinaryPredict>
636 KERBAL_CONSTEXPR14
637 void _K_inplace_merge_ras_iter_loop(RandomAccessIterator first, RandomAccessIterator mid, RandomAccessIterator last, BinaryPredict cmp)
638 {
639 typedef RandomAccessIterator iterator;
640 typedef typename kerbal::iterator::iterator_traits<iterator>::difference_type difference_type;
641
642 while (true) {
643 difference_type left_len(kerbal::iterator::distance(first, mid));
644 difference_type right_len(kerbal::iterator::distance(mid, last));
645
646 if (left_len == 0 || right_len == 0) {
647 return;
648 }
649
650 if (left_len == 1 && right_len == 1) {
651 if (cmp(*mid, *first)) { // *first > *mid
652 kerbal::algorithm::iter_swap(first, mid);
653 }
654 return;
655 }
656
657 iterator lmid(first);
658 iterator rmid(mid);
659
660 if (left_len >= right_len) {
661 lmid = kerbal::iterator::midden_iterator(first, mid);
662 rmid = kerbal::algorithm::lower_bound(mid, last, *lmid, cmp);
663 } else {
664 rmid = kerbal::iterator::midden_iterator(mid, last);
665 lmid = kerbal::algorithm::upper_bound(first, mid, *rmid, cmp);
666 }
667 mid = kerbal::algorithm::rotate(lmid, mid, rmid);
668 _K_inplace_merge_ras_iter_loop(first, lmid, mid, cmp);
669
670 first = mid;
671 mid = rmid; // kerbal::algorithm::inplace_merge(mid, rmid, last, cmp);
672 }
